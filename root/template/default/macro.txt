[%- MACRO link_to(c) BLOCK -%]
<a href="[%- c -%]" target="_blank">[%- c -%]</a>
[%- END -%]
[%-

-%]
[%- MACRO link_or_blank(url, str) BLOCK -%]
[%- IF url %]
                                        <a href="[% url %]" rel="nofollow">[% str %]</a>
[%- ELSE %]
何か取得に失敗しました。。。
[%- END %]
[%- END -%]
[%-

-%]
[%- MACRO date(dt) BLOCK -%]
[%- USE p0 = format('%02d') -%]
[% dt.year %]年[% dt.month %]月[% dt.day %]日 [% dt.hour %]:[% p0(dt.minute) %]
[%- END -%]
